rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper function to check if current user is admin
    function isAdmin() {
      return request.auth != null && 
             exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
             get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin' &&
             get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isActive == true;
    }
    
    // Helper function to check if current user is active
    function isActive() {
      return request.auth != null && 
             exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
             get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isActive == true;
    }
    
    // Users collection
    match /users/{userId} {
      // CRITICAL: Allow users to read their own data FIRST (no helper functions)
      // This must come before admin checks to avoid circular dependency
      allow read: if request.auth != null && request.auth.uid == userId;
      
      // Admins can read all users
      allow read: if isAdmin();
      
      // Only admins can create/update/delete users
      // Allow setup page to create first admin (request.auth == null for initial setup)
      allow create: if request.auth == null || isAdmin();
      allow update: if isAdmin();
      allow delete: if isAdmin();
      
      // Users can update their own profile (limited fields)
      // Note: No isActive() check here to avoid circular dependency
      allow update: if request.auth != null && 
                    request.auth.uid == userId &&
                    // Only allow updating specific fields
                    request.resource.data.diff(resource.data).affectedKeys()
                      .hasOnly(['name', 'unitManager']);
    }
    
    // Strategic planning goals collection
    match /strategic_planning_goals/{goalId} {
      // Users can read/write their own goals
      allow read, write: if request.auth != null && 
                        isActive() &&
                        resource.data.userId == request.auth.uid;
      
      // Admins can read/write all goals
      allow read, write: if isAdmin();
      
      // On create, ensure userId matches authenticated user (unless admin)
      allow create: if request.auth != null && 
                    isActive() &&
                    (request.resource.data.userId == request.auth.uid || isAdmin());
    }
    
    // Data collection (dashboard data)
    match /data/{document=**} {
      // All authenticated active users can read
      allow read: if request.auth != null && isActive();
      
      // Only admins can write
      allow write: if isAdmin();
    }
    
    // Config collection
    match /config/{document=**} {
      // Only admins can access
      allow read, write: if isAdmin();
    }
    
    // Agencies collection
    match /agencies/{document=**} {
      // All authenticated active users can read agencies (needed for dropdowns)
      allow read: if request.auth != null && isActive();
      
      // Only admins can write (add/update/delete agencies)
      allow write: if isAdmin();
    }
    
    // Organizational hierarchy collection
    match /organizational_hierarchy/{document=**} {
      // All authenticated active users can read hierarchy (needed for signup dropdowns)
      allow read: if request.auth != null && isActive();
      
      // Only admins can write (initialize/import hierarchy data)
      allow write: if isAdmin();
    }
  }
}
